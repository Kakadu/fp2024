type ident = string
val equal_ident : ident -> ident -> Ppx_deriving_runtime.bool
val pp_ident : Ppx_deriving_runtime.Format.formatter -> ident -> Ppx_deriving_runtime.unit
val show_ident : ident -> Ppx_deriving_runtime.string
type const =
  1. | Const_bool of bool
    (*

    Boolean constants

    *)
  2. | Const_int of int
    (*

    Integer constants

    *)
  3. | Const_string of string
    (*

    String constants

    *)
  4. | Const_nil
    (*

    *)
val equal_const : const -> const -> Ppx_deriving_runtime.bool
val pp_const : Ppx_deriving_runtime.Format.formatter -> const -> Ppx_deriving_runtime.unit
val show_const : const -> Ppx_deriving_runtime.string
type bin_op =
  1. | Add
    (*

    Addition: 1 + 2

    *)
  2. | Sub
    (*

    Subtraction: 2 - 1

    *)
  3. | Mul
    (*

    Multiplication: 4 * 2

    *)
  4. | Div
    (*

    Division: 8 / 2

    *)
  5. | Mod
    (*

    Modulus: 10 % 5

    *)
  6. | Eq
    (*

    Equality: 5 = 5

    *)
  7. | Con
    (*

    hd::tl

    *)
  8. | Neq
    (*

    Inequality: 4 <> 3

    *)
  9. | Lt
    (*

    Less than: 7 < 10

    *)
  10. | Gt
    (*

    Greater than: 7 > 5

    *)
  11. | Leq
    (*

    Less than or equal to: 10 <= 10

    *)
  12. | Geq
    (*

    Greater than or equal to: 7 >= 7

    *)
  13. | Or
    (*

    Logical OR: true || false

    *)
  14. | And
    (*

    Logical AND: true && false

    *)
val equal_bin_op : bin_op -> bin_op -> Ppx_deriving_runtime.bool
val pp_bin_op : Ppx_deriving_runtime.Format.formatter -> bin_op -> Ppx_deriving_runtime.unit
val show_bin_op : bin_op -> Ppx_deriving_runtime.string
type pattern =
  1. | Pattern_id of ident
  2. | Pattern_const of const
  3. | Pattern_list of pattern * pattern
  4. | Pattern_tuple of pattern list
  5. | Pattern_wild
    (*

    _

    *)
val equal_pattern : pattern -> pattern -> Ppx_deriving_runtime.bool
val pp_pattern : Ppx_deriving_runtime.Format.formatter -> pattern -> Ppx_deriving_runtime.unit
val show_pattern : pattern -> Ppx_deriving_runtime.string
type expr =
  1. | Expr_const of const
    (*

    Represents constant type (string, bool, or int)

    *)
  2. | Expr_var of ident
    (*

    Variable

    *)
  3. | Expr_fun of pattern * expr
    (*

    Function with pattern and type

    *)
  4. | Expr_app of expr * expr
    (*

    Function application

    *)
  5. | Expr_if of expr * expr * expr
    (*

    Conditional operator

    *)
  6. | Expr_let_in of bool * ident * expr * expr
    (*

    Let binding with in clause

    *)
  7. | Expr_bin_op of bin_op * expr * expr
    (*

    Binary operation, e.g., 2 + 2 - 3

    *)
  8. | Expr_match of expr * (pattern * expr) list
    (*

    match x_1 with | x_2 -> ...

    *)
  9. | Expr_list of expr * expr
    (*

    1;2;3

    *)
  10. | Expr_tuple of expr list
    (*

    (1,2,3)

    *)
val equal_expr : expr -> expr -> Ppx_deriving_runtime.bool
val pp_expr : Ppx_deriving_runtime.Format.formatter -> expr -> Ppx_deriving_runtime.unit
val show_expr : expr -> Ppx_deriving_runtime.string
type struct_prog =
  1. | Let of bool * ident * expr
  2. | Expression of expr
val pp_struct_prog : Ppx_deriving_runtime.Format.formatter -> struct_prog -> Ppx_deriving_runtime.unit
val show_struct_prog : struct_prog -> Ppx_deriving_runtime.string
