type typ =
  1. | Prim of string
    (*

    Available ground types

    *)
  2. | Ty_var of int
    (*

    a, b types

    *)
  3. | Arrow of typ * typ
    (*

    Function type: 'a -> 'a

    *)
  4. | List_typ of typ
    (*

    List type: int list

    *)
  5. | Tuple_typ of typ list
    (*

    Typle type: int, string means (int, string)

    *)
