and expr =
  1. | Const of literal
    (*

    Int, Bool, String, Unit, Null

    *)
  2. | Tuple of expr * expr * expr list
    (*

    (1, "Hello world", true)

    *)
  3. | List of expr list
    (*

    , 1;2;3

    *)
  4. | Variable of ident
    (*

    x, y

    *)
  5. | Unary_expr of unary_operator * expr
    (*

    -x

    *)
  6. | Bin_expr of binary_operator * expr * expr
    (*

    1 + 2, 3 ||| 12, hd :: tl

    *)
  7. | If_then_else of expr * expr * expr option
    (*

    if n % 2 = 0 then "Even" else "Odd"

    *)
  8. | Lambda of pattern * pattern list * expr
    (*

    fun x y -> x + y

    *)
  9. | Apply of expr * expr
    (*

    sum 1

    *)
  10. | Function of case * case list
    (*

    function | p1 -> e1 | p2 -> e2 | ... |

    *)
  11. | Match of expr * case * case list
    (*

    match x with | p1 -> e1 | p2 -> e2 | ...

    *)
  12. | LetIn of is_recursive * let_bind * let_bind list * expr
    (*

    let rec f x = if (x <= 0) then x else g x and g x = f (x-2) in f 3

    (*

    int option

    *)
  14. | EConstraint of expr * TypedTree.typ
