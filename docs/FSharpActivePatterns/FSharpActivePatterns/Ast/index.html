type ident =
  1. | Ident of string
type literal =
  1. | Int_lt of int
  2. | Bool_lt of bool
  3. | String_lt of string
  4. | Unit_lt
type binary_operator =
  1. | Binary_equal
  2. | Binary_unequal
  3. | Binary_less
  4. | Binary_less_or_equal
  5. | Binary_greater
  6. | Binary_greater_or_equal
  7. | Binary_add
  8. | Binary_subtract
  9. | Binary_multiply
  10. | Logical_or
  11. | Logical_and
  12. | Binary_divide
  13. | Binary_or_bitwise
  14. | Binary_xor_bitwise
  15. | Binary_and_bitwise
type unary_operator =
  1. | Unary_plus
  2. | Unary_minus
  3. | Unary_negative
type is_recursive =
  1. | Nonrec
  2. | Rec
type expr =
  1. | Const of literal
  2. | Tuple of expr list
  3. | Variable of ident
  4. | Bin_expr of binary_operator * expr * expr
    (*

    operator, first operand, second operand

    *)
  5. | If_then_else of expr * expr list * expr list option
    (*

    condition, then body, else body

    *)
  6. | Function_def of is_recursive * string option * expr list * expr list
    (*

    rec, name, args, body

    *)
  7. | Function_call of string * expr list
    (*

    name, args

    *)
  8. | Let of ident * expr
    (*

    name, value

    *)
  9. | LetIn of ident * expr * expr
    (*

    name, value, inner value

    *)
