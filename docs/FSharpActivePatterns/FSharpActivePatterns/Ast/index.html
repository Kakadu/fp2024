type expr =
  1. | Const of literal
    (*

    Int, Bool, String, Unit, Null

    *)
  2. | Tuple of expr list
    (*

    (1, "Hello world", true)

    *)
  3. | List_expr of expr * expr
    (*
    [ 1, 2, 3 ]
    *)
  4. | Variable of ident
    (*

    x, y

    *)
  5. | Unary_expr of unary_operator * expr
  6. | Bin_expr of binary_operator * expr * expr
    (*

    1 + 2, 3 ||| 12

    *)

    (*

    if n % 2 = 0 then "Even" else "Odd"

    *)

    rec, name, args, body

    *)
  9. | Function_call of expr * expr list
    (*

    sum 1

    *)
  10. | Match of expr * (pattern * expr) list
    (*

    match x with | x -> ... | y -> ...

    *)

    (*

    let x = 5 in x * y
