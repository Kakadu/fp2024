type expression =
  1. | Expr_var of ident
    (*

    e.g. x

    *)
  2. | Expr_const of constant
    (*

    e.g. 3, true, "string"

    *)
  3. | Expr_option of expression option
    (*

    e.g. Some (x-3), None

    *)
  4. | Expr_list_sugar of expression list
    (*

    e.g. [4;5;6]

    *)
  5. | Expr_list_construct of expression list
    (*

    e.g. 4::5::6::[]

    *)
  6. | Expr_tuple of expression list
    (*

    e.g. (x,5+y))

    *)
  7. | Expr_binary_op of binary_op * expression * expression
    (*

    e.g. 2+3 , 23<x , a||b

    *)
  8. | Expr_if_then_else of expression * expression * expression
    (*

    e.g. if x then 1 else 0

    *)
  9. | Expr_match_with of expression * (pattern * expression) list
    (*

    e.g. match x with | a::[] -> true | _ -> false

    *)
  10. | Expr_construct_in of let_binding * expression
    (*

    e.g. let x = 4 in x

    *)
  11. | Expr_anonym_fun of pattern list * expression
    (*

    e.g. fun a b -> a + b

    *)
  12. | Expr_function_fun of (pattern * expression) list
    (*

    e.g. function | (a,b) -> a+b | _ -> 0

    *)
  13. | Expr_application of expression * expression list
    (*

    e.g. f a b (fun a -> a*3) 5 (function | true -> 1 | false -> 0) x

    *)
  14. | Typed_expression of ttype * expression
    (*

    e.g. (x : int)

    *)
