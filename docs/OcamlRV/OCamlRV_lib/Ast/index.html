type identifier = string
val pp_identifier : Ppx_deriving_runtime.Format.formatter -> identifier -> Ppx_deriving_runtime.unit
val show_identifier : identifier -> Ppx_deriving_runtime.string
type binary_operator =
  1. | Add
  2. | Sub
  3. | Mul
  4. | Div
  5. | Lt
  6. | Gt
  7. | Eq
  8. | Neq
  9. | Lte
  10. | Gte
  11. | And
  12. | Or
val pp_binary_operator : Ppx_deriving_runtime.Format.formatter -> binary_operator -> Ppx_deriving_runtime.unit
val show_binary_operator : binary_operator -> Ppx_deriving_runtime.string
type unary_operator =
  1. | UnaryPlus
  2. | UnaryMinus
  3. | UnaryNeg
val pp_unary_operator : Ppx_deriving_runtime.Format.formatter -> unary_operator -> Ppx_deriving_runtime.unit
val show_unary_operator : unary_operator -> Ppx_deriving_runtime.string
type rec_flag =
  1. | NonRec
  2. | Rec
val pp_rec_flag : Ppx_deriving_runtime.Format.formatter -> rec_flag -> Ppx_deriving_runtime.unit
val show_rec_flag : rec_flag -> Ppx_deriving_runtime.string
type placement =
  1. | Global
  2. | Inline of identifier
val pp_placement : Ppx_deriving_runtime.Format.formatter -> placement -> Ppx_deriving_runtime.unit
val show_placement : placement -> Ppx_deriving_runtime.string
type literal =
  1. | IntLiteral of int
  2. | BoolLiteral of bool
  3. | StringLiteral of string
  4. | UnitLiteral
  5. | NilLiteral
val pp_literal : Ppx_deriving_runtime.Format.formatter -> literal -> Ppx_deriving_runtime.unit
val show_literal : literal -> Ppx_deriving_runtime.string
type pattern =
  1. | PAny
  2. | PLiteral of literal
  3. | PVar of identifier
  4. | PTuple of pattern list
  5. | PCons of pattern * pattern
  6. | PPoly of identifier * pattern option
val pp_pattern : Ppx_deriving_runtime.Format.formatter -> pattern -> Ppx_deriving_runtime.unit
val show_pattern : pattern -> Ppx_deriving_runtime.string
type expression =
  1. | ExprVariable of identifier
  2. | ExprLiteral of literal
  3. | ExprBinOperation of binary_operator * expression * expression
  4. | ExprUnOperation of unary_operator * expression
  5. | ExprIf of expression * expression * expression option
  6. | ExprMatch of expression * case list
  7. | ExprLet of rec_flag * pattern list * expression list
  8. | ExprApply of expression * expression
  9. | ExprTuple of expression list
  10. | ExprCons of expression * expression
  11. | ExprPoly of identifier * expression option
  12. | ExprFun of pattern * pattern list * expression
and case = pattern * expression
val pp_expression : Ppx_deriving_runtime.Format.formatter -> expression -> Ppx_deriving_runtime.unit
val show_expression : expression -> Ppx_deriving_runtime.string
val pp_case : Ppx_deriving_runtime.Format.formatter -> case -> Ppx_deriving_runtime.unit
val show_case : case -> Ppx_deriving_runtime.string
type program = expression list
val pp_program : Ppx_deriving_runtime.Format.formatter -> program -> Ppx_deriving_runtime.unit
val show_program : program -> Ppx_deriving_runtime.string
