type value =
  1. | ValInt of Base.int
  2. | ValStr of Base.string
  3. | ValBool of Base.bool
  4. | ValUnit
  5. | ValList of value Base.list
  6. | ValTup of value * value * value Base.list
  7. | ValFun of Ast.rec_state * Ast.pat * Ast.pat Base.list * Ast.expr * env
  8. | ValBuiltIn of Base.string
  9. | ValOption of value Base.option
and env = (Ast.id, value, Base.String.comparator_witness) Base.Map.t

  1. | NoVariable of Base.string
  2. | TypeError
  3. | PatternMatchingFail
