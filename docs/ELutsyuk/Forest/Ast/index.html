type pat =
  1. | PatAny

  2. | PatConst of const

    Matches a constant value, e.g. 42, true.

    (*

    Matches a cons pattern for lists, e.g. x::y.

  4. | PatVar of id
    (*

    Matches any value and binds it to a variable, e.g. x.

    *)
  5. | PatTup of pat * pat * pat list
    (*

    Matches tuples, e.g. x, y, a, b, c.

    *)

  7. | PatType of pat * TypesTree.typ

type expr =
  1. | Var of id
    (*

    Variable reference, e.g. x, my_var.

    *)
  2. | Const of const
    (*

    Constant value, e.g. 42, "hello", true.

    *)
  3. | Unary of unary * expr
  4. | BinOp of binop * expr * expr
    (*

    Binary operation, e.g. x + y, a >= b.

    *)
  5. | Option of expr option
    (*

    Options, e.g. Some("meow"), None.

    *)
  6. | Let of rec_state * let_binding * let_binding list * expr
    (*

    let expression, e.g. let x = 5 in e

    *)
  7. | App of expr * expr
    (*

    Function application, e.g. e1 e2, (fun x -> x) 42.

    *)
  8. | Fun of pat * pat list * expr
    (*

    Function definition, e.g. fun p -> e.

    *)
  9. | Branch of expr * expr * expr
    (*

    Conditional expression, e.g. if e1 then e2 else e3.

    *)
  10. | Tup of expr * expr * expr list
    (*

    Tuple expression, e.g. (e1, e2), (x, y, z).

    *)
  11. | List of expr list
    (*

    List expression, e.g. [], [e1; e2; e3].

    *)
  12. | Type of expr * TypesTree.typ

Represents expressions in the language.
