type error =
  1. | UnboundVariable of string
  2. | TypeMissmatch
  3. | DivisionByZero
type value =
  1. | VInt of int
  2. | VString of string
  3. | VBool of bool
  4. | VTuple of value * value * value list
  5. | VFun of Ast.rec_flag * Ast.pattern * Ast.expression * environment
  6. | VFunMutual of Ast.rec_flag * Ast.pattern * Ast.expression * environment
  7. | VList of value list
  8. | VOption of value option
  9. | VUnit
  10. | VPrintInt
and environment = (string, value, Base.String.comparator_witness) Base.Map.t
val interpret : Ast.program -> ((string, value, Base.String.comparator_witness) Base.Map.t, error) Stdlib.result
val pp_error : error -> string
