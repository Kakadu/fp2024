Module Ast.Pattern

type t =
  1. | Pat_constraint of t * TypeExpr.t
    (*

    Pattern (P : T)

    *)
  2. | Pat_any
    (*

    The pattern _.

    *)
  3. | Pat_var of ident
    (*

    A variable pattern such as x

    *)
  4. | Pat_constant of Constant.t
    (*

    Patterns such as 52, 'w', "uwu"

    *)
  5. | Pat_tuple of t List2.t
    (*

    Patterns (P1, ..., Pn).

    *)
  6. | Pat_construct of ident * t Base.option
    (*

    Pat_construct(C, args) represents:

    • C when args is None,
    • C P when args is Some (P)
    • C (P1, ..., Pn) when args is Some (Pat_tuple [P1; ...; Pn])
    *)
