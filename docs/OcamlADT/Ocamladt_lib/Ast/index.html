type pattern =
  1. | Pat_any
    (*

    The pattern _.

    *)
  2. | Pat_var of ident
    (*

    A variable pattern such as x

    *)
  3. | Pat_constant of constant
    (*

    Patterns such as 52, 'w', "uwu"

    *)
  4. | Pat_tuple of pattern list2
    (*

    Patterns (P1, ..., Pn).

    *)
  5. | Pat_construct of ident * pattern option
    (*

    Pat_construct(C, args) represents:

    • C when args is None,
    • C P when args is Some (P)
    • C (P1, ..., Pn) when args is Some (Pat_tuple [P1; ...; Pn])
    *)
val equal_pattern : pattern -> pattern -> Ppx_deriving_runtime.bool
val pp_pattern : Ppx_deriving_runtime.Format.formatter -> pattern -> Ppx_deriving_runtime.unit
val show_pattern : pattern -> Ppx_deriving_runtime.string

type expression =
  1. | Exp_ident of ident
    (*

    Identifiers such as x

    *)
  2. | Exp_constant of constant
    (*

    Expressions constant such as 1, 'a', "true"*

    *)
  3. | Exp_tuple of expression list2
    (*

    Expressions (E1, E2, ..., En)

    *)
  4. | Exp_function of case list1
    (*

    Exp_function (P1, [P2; ...; Pn]) represents function P1 | ... | Pn

    *)
  5. | Exp_fun of pattern list1 * expression
    (*

    Exp_fun (P1, [P2; ...; Pn], E) represents: fun P1 ... Pn -> E

    *)
  6. | Exp_apply of expression * expression list1
    (*

    Pexp_apply(E0, [E1 ; ... ; En]) represents E0 E1 ... En

    *)
  7. | Exp_match of expression * case list1
    (*

    match E0 with P1 -> E1 |[@@deriving eq, show { with_path = false }] | Pn -> En

    *)
  8. | Exp_if of expression * expression * expression option
    (*

    if E1 then E2 else E3

    *)
  9. | Exp_let of rec_flag * value_binding list1 * expression
    (*

    Exp_let(flag, [(P1,E1) ; ... ; (Pn,En)], E) represents:

    • let P1 = E1 and ... and Pn = EN in E when flag is Nonrecursive,
    • let rec P1 = E1 and ... and Pn = EN in E when flag is Recursive.
    *)
  10. | Exp_construct of ident * expression option
    (*

    Exp_construct(C, exp) represents:

    • C when exp is None,
    • C E when exp is Some E,
    • C (E1, ..., En) when exp is Some (Exp_tuple[E1;...;En])
    *)
and value_binding = {
  1. pat : pattern;
  2. expr : expression;
}
and case = {
  1. left : pattern;
  2. right : expression;
}

  1. | Type_arrow of type_expr * type_expr

    Type_arrow(T1, T2) represents: T1 -> T2

    Type_constr(ident, l) represents:

    • tconstr when l=[],
    • T tconstr when l=[T],
    • (T1, ..., Tn) tconstr when l=[T1 ; ... ; Tn].
    *)

type structure_item =
  1. | Str_eval of expression
  2. | Str_value of rec_flag * value_binding list1
    (*

    Str_value(rec, [(P1, E1 ; ... ; (Pn, En))]) represents:

    • let P1 = E1 and ... and Pn = EN when rec is Nonrecursive,
    • let rec P1 = E1 and ... and Pn = EN when rec is Recursive.
    *)
  3. | Str_adt of ident * (ident * type_expr) list1
    (*

    Str_type(C0, [(C1, [(T11; T12; ... ; T1n_1)]); (C2, [(T21;T22; ... ; T2n_2)]); ... ; (Cm, [(Tm1;Tm2; ... ; Tmn_n)]) ]) represents:

    type C0 = | C1 of T11 * ... * T1n_1 | ... | Cm of Tm1 * ... * Tmn_n

    n_i: n_i >= 0 Invariant: m > 0

    *)
val equal_structure_item : structure_item -> structure_item -> Ppx_deriving_runtime.bool
val pp_structure_item : Ppx_deriving_runtime.Format.formatter -> structure_item -> Ppx_deriving_runtime.unit
val show_structure_item : structure_item -> Ppx_deriving_runtime.string
