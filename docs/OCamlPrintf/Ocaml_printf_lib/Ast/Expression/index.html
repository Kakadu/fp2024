Module Ast.Expression

type t =
  1. | Exp_ident of ident
    (*

    Identifier such as x

    *)
  2. | Exp_constant of constant
    (*

    ts constant such as 1, 'a', "true"

    *)
  3. | Exp_let of rec_flag * t value_binding * t value_binding list_ * t
    (*

    Exp_let(flag, [(P1, E1); ... ; (Pn, En)], E) represents:

    • let P1 = E1 and ... and Pn = En in E when flag is Nonrecursive,
    • let rec P1 = E1 and ... and Pn = En in E when flag is Recursive.
    *)
  4. | Exp_fun of pattern * pattern list_ * t
    (*

    Exp_fun([P1; ... ; Pn], E) represents fun P1 ... Pn -> E

    *)
  5. | Exp_apply of t * t * t list
    (*

    Exp_apply(E0, [E1; ... ; En]) represents E0 E1 ... En

    *)
  6. | Exp_match of t * t case * t case list_
    (*

    match E0 with P1 -> E1 | ... | Pn -> En

    *)
  7. | Exp_tuple of t * t * t list_
    (*

    ts (E1, ... , En)

    *)
  8. | Exp_construct of ident * t option
    (*

    Exp_construct(C, exp) represents:

    • C when exp is None,
    • C E when exp is Some E,
    • C (E1, ... , En) when exp is Some (Exp_tuple[E1; ... ; En])
    *)
  9. | Exp_ifthenelse of t * t * t option
    (*

    if E1 then E2 else E3

    *)
  10. | Exp_sequence of t * t
    (*

    E1; E2

    *)
  11. | Exp_constraint of t * core_type
    (*

    (E : T)

    *)
val show : t -> ident
