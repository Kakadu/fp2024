type expr =
  1. | Expr_nil
    (*

    A value of an unitialized channel or function: nil

    *)
  2. | Expr_const of const
    (*

    Constants such as 5, "hi", false

    *)
  3. | Expr_ident of ident
    (*

    An identificator for a variable such as x

    *)
  4. | Expr_index of ident * expr
    (*

    An access to an array element by its index: my_array[i]

    *)
  5. | Expr_bin_oper of bin_oper
    (*

    Binary operations such as a + b, x || y

    *)
  6. | Expr_un_oper of unary_oper
    (*

    Unary operations such as !z, -f

    *)
  7. | Expr_anon_func of (ident * type' option) list option * type' list option * stmt
    (*

    An anonymous function such as func() {}, func(a int, b int) int { return a + b }

    *)
  8. | Expr_call of expr * expr list option
    (*

    function calls such as: my_func(arg1, arg2), c()()(), func() { println("hello") }()

    *)

Expressions that can be assigned to a variable or put in "if" statement

and bin_oper =
  1. | Bin_sum of expr * expr
    (*

    Binary sum: 1 + 1

    *)
  2. | Bin_multiply of expr * expr
    (*

    Binary multiplication: a * 5

    *)
  3. | Bin_subtract of expr * expr
    (*

    Binary subtraction: func1(x) - func2(y)

    *)
  4. | Bin_divide of expr * expr
    (*

    Binary divison: 7 / 3

    *)
  5. | Bin_modulus of expr * expr
    (*

    Binary division by modulus: 123 % 10

    *)
  6. | Bin_equal of expr * expr
    (*

    Binary check for equality: result == 25

    *)
  7. | Bin_not_equal of expr * expr
    (*

    Binary check for inequlity: i != n

    *)
  8. | Bin_greater of expr * expr
    (*

    Binary "greater than": sum > minimum

    *)
  9. | Bin_greater_equal of expr * expr
    (*

    Binary "greater than or equal": b >= a

    *)
  10. | Bin_less of expr * expr
    (*

    Binary "less than": sum < maximum

    *)
  11. | Bin_less_equal of expr * expr
    (*

    Binary "less than or equal": a <= b

    *)
  12. | Bin_and of expr * expr
    (*

    Binary "and": ok && flag

    *)
  13. | Bin_or of expr * expr
    (*

    Binary "or": is_online || is_friend

    *)

Binary operations

and unary_oper =

    (*

    Unary negation: !x

    *)
  2. | Unary_plus of expr
    (*

    Unary plus: +a

    *)
  3. | Unary_minus of expr
    (*

Unary operations

and stmt =
  1. | Stmt_empty
    (*

    Empty statement, i. e. empty function body

    *)
  2. | Stmt_var_decl of ident list * type' option * expr list option
    (*

    Declaration of a variable inside of a function such as: var array []int, flag := true, var a int = 5

    *)
  3. | Stmt_assign of ident list * expr list
    (*

    Assignment to a variable such as a = 3, a, b = 4, 5

    *)
  4. | Stmt_incr of ident
    (*

    An increment of a variable: a++

    *)
  5. | Stmt_decr of ident
    (*

    A deincrement of a variable: a--

    *)
  6. | Stmt_if of stmt option * expr * stmt * stmt option
    (*

    An if statement such as: if a := 5; a >= 4 { do() } else { do_else() }

    *)
  7. | Stmt_for of stmt option * expr option * stmt option * stmt
    (*

    A for statement such as: for i := 0; i < n; i++ { do() }

    *)
  8. | Stmt_range of ident * ident option * ident * stmt
    (*

    For with range statement such as: for i, elem := range array { check(elem) }

    *)
  9. | Stmt_break
    (*

    Break statemnet: break

    *)
  10. | Stmt_continue
    (*

    Continue statement: continue

    *)
  11. | Stmt_return of expr option
    (*

    Return statement such as return some_expr or return

    *)
  12. | Stmt_defer of expr
    (*

    Defer statement such as defer close_file()

    *)
  13. | Stmt_block of stmt list
    (*

    Block of statements in curly braces

    *)
  14. | Stmt_call of expr * expr list option
    (*

    The same as Expr_call in expr type

    *)
  15. | Stmt_channel_send of ident * expr
    (*

    Channel send operation c <- true

    *)
  16. | Stmt_channel_recieve of ident * expr
    (*

    Channel recieve operation z := <-c

    *)
  17. | Stmt_go of expr
    (*

    Goroutine statement: go func(ch chan<- bool)

    *)

and func_decl = ident * (ident * type' option) list option * type' list option * stmt
