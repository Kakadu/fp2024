type expr =
  1. | Expr_nil
    (*

    A value of an unitialized channel or function: nil

    *)
  2. | Expr_const of const
    (*

    Constants such as 5, "hi", false

    *)
  3. | Expr_array of type' * expr list option
    (*

    Arrays such as [3]int{3, get_four()}

    *)
  4. | Expr_ident of ident
    (*

    An identificator for a variable such as x

    *)
  5. | Expr_index of expr * expr
    (*

    An access to an array element by its index such as: my_array[i], get_array(1)[0]

    *)
  6. | Expr_bin_oper of bin_oper * expr * expr
    (*

    Binary operations such as a + b, x || y

    *)
  7. | Expr_un_oper of unary_oper * expr
    (*

    Unary operations such as !z, -f

    *)
  8. | Expr_anon_func of anon_func
    (*

    See anon_func type

    *)
  9. | Expr_call of func_call
    (*

    See func_call type

    *)

Expressions that can be assigned to a variable or put in "if" statement

and anon_func = {
  1. args : (ident * type' option) list option;
    (*

    arguments

    *)
  2. return_types : (ident option * type' option) list option;
    (*

    return types, optional var names

    *)
  3. body : stmt option;
    (*

    function body

    *)
}

An anonymous functions such as: func() {}, func(a, b int) (sum int) { sum = a + b; return } func(s1 string, s2 string) [2]string { return [2]string{s1,s2} }

and func_call = expr * expr list option

function calls such as: my_func(arg1, arg2), c()()(), func() { println("hello") }()

and stmt =
  1. | Stmt_var_decl of var_decl
    (*

    See var_decl type

    *)
  2. | Stmt_assign of ident list * expr list
    (*

    Assignment to a variable such as a = 3, a, b = 4, 5

    *)
  3. | Stmt_incr of ident
    (*

    An increment of a variable: a++

    *)
  4. | Stmt_decr of ident
    (*

    A decrement of a variable: a--

    *)
  5. | Stmt_if of stmt option * expr * stmt option * stmt option
    (*

    An if statement such as: if a := 5; a >= 4 { do() } else { do_else() }

    *)
  6. | Stmt_for of stmt option * expr option * stmt option * stmt option
    (*

    A for statement such as: for i := 0; i < n; i++ { do() }

    *)
  7. | Stmt_range of ident * ident option * expr * stmt option
    (*

    For with range statement such as: for i, elem := range array { check(elem) }

    *)
  8. | Stmt_break
    (*

    Break statement: break

    *)
  9. | Stmt_continue
    (*

    Continue statement: continue

    *)
  10. | Stmt_return of expr option
    (*

    Return statement such as return some_expr or return

    *)
  11. | Stmt_block of stmt list
    (*

    Block of statements in curly braces

    *)
  12. | Stmt_chan_send of ident * expr
    (*

    Channel send operation c <- true

    *)
  13. | Stmt_chan_recieve of ident * expr
    (*

    Channel recieve operation z := <-c

    *)
  14. | Stmt_call of func_call
    (*

    See func_call type

    *)
  15. | Stmt_defer of func_call
    (*

    See func_call type

    *)
  16. | Stmt_go of func_call
    (*

    See func_call type

    *)

Statement, a syntactic unit of imperative programming
