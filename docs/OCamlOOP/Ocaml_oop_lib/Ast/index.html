type expression =
  1. | Exp_ident of ident
    (*

    Identifier such as x

    *)
  2. | Exp_constant of constant
    (*

    Constants such as 24, '3.14', "true", ...

    *)
  3. | Exp_let of decl * expression
    (*

    Exp_let(flag, [(P1,E1) ; ... ; (Pn,En)], E) represents:

    • let P1 = E1 and ... and Pn = EN in E when flag is Nonrecursive,
    • let rec P1 = E1 and ... and Pn = EN in E when flag is Recursive.
    *)
  4. | Exp_unary of unary_operator * expression
    (*

    Unary operators such as -E, not E

    *)
  5. | Exp_binary of expression * binary_operator * expression
    (*

    Binary operators such as E1 + E2, E1 && E2

    *)
  6. | Exp_ifthenelse of expression * expression * expression option
    (*

    if E1 then E2 else E3

    *)
  7. | Exp_send of expression * ident
    (*

    E # m

    *)
  8. | Exp_list of expression * expression
    (*

    The expression such as E1::E2 This also represents lists E1; ... En via E

    *)
  9. | Exp_tuple of expression list
    (*

    Tuples such as (E1, ..., En)

    *)
  10. | Exp_function of pattern list * expression
    (*

    Function such as fun P -> E

    *)
  11. | Exp_apply of expression * expression list
    (*

    Exp_apply(E0, [E1; ...; En]) represents E0 E1 ... En

    *)
  12. | Exp_object of obj
    (*

    object ... end

    *)
  13. | Exp_override of (ident * expression) list
    (*

    {< x1 = E1; ...; xn = En >}

    *)
  14. | Exp_match of expression * (pattern * expression) list
    (*

    Exp_match(E, [C1; ...; Cn]) represents match E with C1 | ... | Cn

    *)
  15. | Exp_construct of ident * expression option
    (*

    Exp_construct(C, exp) represents:

    • C when exp is None,
    • C E when exp is Some E,
    • C (E1, ..., En) when exp is Some (Exp_tuple[E1;...;En])
    *)
and decl = {
