type pattern =
  1. | PVar of id
    (*

    variable pattern, e.g. x

    *)
  2. | PConst of const
    (*

    pattern of a constant

    *)
  3. | PTuple of pattern * pattern * pattern list
    (*

    patterns (P0, .., Pn), n >= 2

    *)
  4. | PAny
    (*

    wildcard pattern '_'

    *)
  5. | PList of pattern list
    (*

    patterns P0; ..; Pn, n >= 0

    *)
  6. | PCons of pattern * pattern
    (*

    P0 :: P1 :: .. :: Pn, n>= 2

    *)
  7. | POption of pattern option
    (*

    Some p, None

    *)
  8. | PConstraint of pattern * Typedtree.ty
    (*

    P : T

    *)

type expr =
  1. | Econst of const
    (*

    constants, e.g. 10, "meow", true

    *)
  2. | Evar of id
    (*

    identifiers, e.g. "x", "f"

    *)
  3. | Eif_then_else of expr * expr * expr option
    (*

    if E0 then E1 else E2; else expression is optional

    *)
  4. | Eoption of expr option
    (*

    option type, Some e, None

    *)
  5. | Etuple of expr * expr * expr list
    (*

    expressions (E0, .., En), n >= 2

    *)
  6. | Elist of expr list
    (*

    expressions E0; ..; En, n >= 0

    *)
  7. | Ebin_op of bin_op * expr * expr
    (*

    E0 bin_op E1, e.g. 1 + 3

    *)
  8. | Ematch of expr * case * case list
    (*

    match E with P1 -> E1 ... Pn -> Pn

    *)
  9. | Efunction of case * case list
    (*

    function P1 -> E1 ... Pn -> Pn

    *)
  10. | Eun_op of un_op * expr
    (*

    E0 un_op E1, e.g. Negative 2, Not true

    *)
  11. | Elet of rec_flag * value_binding * value_binding list * expr
    (*

    let (rec) P1 = E1 and P2 = E2 and ... and Pn = En in E, e.g. let x = 5 in x - 10

    *)
  12. | Efun_application of expr * expr
    (*

    E0 E1, e.g. f x

    *)
  13. | Efun of pattern * pattern list * expr
    (*

    anonymous functions, e.g. fun x y -> x + 1 - y, arguments num >= 1

    *)
  14. | Econstraint of expr * Typedtree.ty
    (*

    E : T

    *)
and case =
  1. | Ecase of pattern * expr
    (*

    pattern-matching case, e.g. | P -> E

    *)
and value_binding =
  1. | Evalue_binding of pattern * expr
    (*

    P = E

    *)
val show_expr : expr -> string
val show_case : case -> string
val show_value_binding : value_binding -> string
type structure_item =

    (*

    let (rec) P1 = E1 and P2 = E2 and ... and Pn = En e.g. let x = 5

    *)
val show_structure_item : structure_item -> string
type structure = structure_item list
val show_structure : structure -> string
