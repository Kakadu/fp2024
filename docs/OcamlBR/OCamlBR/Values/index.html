type builtin =
  1. | BInt of int -> unit
  2. | BString of string -> unit
type value =
  1. | VInt of int
  2. | VBool of bool
  3. | VString of string
  4. | VUnit
  5. | VList of value list
  6. | VTuple of value * value * value list
  7. | VFun of Ast.rec_flag * Ast.pattern * Ast.pattern list * Ast.expr * environment
  8. | VOption of value option
  9. | VBuiltin of builtin * environment
  10. | VFunction of Ast.case * Ast.case list
and environment = (string, value, Base.String.comparator_witness) Base.Map.t
val pp_value : Stdlib.Format.formatter -> value -> unit
type error = [
  1. | `Division_by_zero
  2. | `Unbound_variable of string
  3. | `Pattern_matching_failure
  4. | `Type_error
  5. | `Ill_left_hand_side of string
]
val pp_error : Stdlib.Format.formatter -> error -> unit
