type expression =
  1. | ExprVariable of identifier
    (*

    x | y | z

    *)
  2. | ExprConstant of constant
    (*

    123 | true | "string"

    *)
  3. | ExprBinOperation of binary_operator * expression * expression
    (*

    1 + 1 | 2 * 2

    *)
  4. | ExprUnOperation of unary_operator * expression
    (*

    -x | not true

    *)
  5. | ExprIf of expression * expression * expression option
    (*

    if expr1 then expr2 else expr3

    *)
  6. | ExprMatch of expression * case * case list
    (*

    match e with p_1 -> e_1 |...| p_n -> e_n

    *)
  7. | ExprLet of rec_flag * binding * binding list * expression
    (*

    ExprLet(rec_flag, (p_1, e_1), [(p_2, e_2) ; ... ; (p_n, e_n)], e)

    *)
  8. | ExprApply of expression * expression
    (*

    fact n

    *)
  9. | ExprTuple of expression * expression * expression list
    (*

    1, 2, 3

    *)
  10. | ExprList of expression * expression list
  11. | ExprCons of expression * expression
    (*

    t::tl

    *)
  12. | ExprFun of pattern * expression
    (*

    fun p -> e

    *)
  13. | ExprOption of expression option
