type binder = int
type ty =
  1. | TPrimitive of string
    (*

    int, string, bool, unit

    *)
  2. | TVar of binder
    (*

    Type variable

    *)
  3. | TArrow of ty * ty
    (*

    Function type

    *)
  4. | TTuple of ty list
    (*

    Tuple type

    *)
  5. | TList of ty
    (*

    List type

    *)
val arrow : ty -> ty -> ty
val int_type : ty
val bool_type : ty
val string_type : ty
val unit_type : ty
val tuple_type : ty list -> ty
val list_type : ty -> ty
val pp_type : Stdlib.Format.formatter -> ty -> unit
type error = [
  1. | `Occurs_check
  2. | `No_variable of Ast.identifier
  3. | `Unification_failed of ty * ty
  4. | `Pattern_matching_error
  5. | `Not_implemented
]
val pp_error : Stdlib.Format.formatter -> error -> unit
