type rational_exp =
  1. | Integer of int
    (*

    Integer exponent: 2

    *)
  2. | Rational of int * int
    (*

    Rational exponent: 3/2

    *)
  3. | Negate of rational_exp
    (*

    Negation of exponent: -1, -(1/2)

    *)
  4. | Paren of rational_exp
    (*

    Parentheses around exponent: (5), (-2/3)

    *)
val pp_rational_exp : Ppx_deriving_runtime.Format.formatter -> rational_exp -> Ppx_deriving_runtime.unit
val show_rational_exp : rational_exp -> Ppx_deriving_runtime.string
type measure =
  1. | Measure_ident of string
    (*

    Measure identificator: <m>

    *)
  2. | Measure_prod of measure * measure
    (*

    Measure product: <sec * h>, <kg m>

    *)
  3. | Measure_div of measure * measure
    (*

    Measure division: <m / sec>

    *)
  4. | Measure_pow of measure * rational_exp
    (*

    Measure to the rational power: <cm^3>

    *)
val pp_measure : Ppx_deriving_runtime.Format.formatter -> measure -> Ppx_deriving_runtime.unit
val show_measure : measure -> Ppx_deriving_runtime.string
type measure_val =
  1. | Mval_int of int
    (*

    Integer numbers with units of measure

    *)
  2. | Mval_float of float
    (*

    Real numbers with units of measure

    *)
val pp_measure_val : Ppx_deriving_runtime.Format.formatter -> measure_val -> Ppx_deriving_runtime.unit
val show_measure_val : measure_val -> Ppx_deriving_runtime.string
type constant =
  1. | Const_bool of bool
    (*

    Boolean constants true and false

    *)
  2. | Const_int of int
    (*

    Integer constants: 1

    *)
  3. | Const_char of char
    (*

    Char constants: 'a'

    *)
  4. | Const_string of string
    (*

    String constants: "foo"

    *)
  5. | Const_float of float
    (*

    Float constants: 3.14, 1e+5, 5.9E-3

    *)
  6. | Const_measure of measure_val * measure
    (*

    Measure constants: 5.0<cm>, 3<kg>

    *)
val pp_constant : Ppx_deriving_runtime.Format.formatter -> constant -> Ppx_deriving_runtime.unit
val show_constant : constant -> Ppx_deriving_runtime.string
type pattern =
  1. | Pattern_wild
    (*

    Wildcard patterns _

    *)
  2. | Pattern_ident of string
    (*

    Identificator name patterns: x

    *)
  3. | Pattern_const of constant
    (*

    Constant patterns: 1, 'a', "foo", 3.14, 5.0<cm>

    *)
  4. | Pattern_tuple of pattern list
    (*

    Tuple patterns: (P1; ..., Pn) Invariant: n >= 2

    *)
  5. | Pattern_or of pattern * pattern
    (*

    Or patterns: P1 | P2

    *)
val pp_pattern : Ppx_deriving_runtime.Format.formatter -> pattern -> Ppx_deriving_runtime.unit
val show_pattern : pattern -> Ppx_deriving_runtime.string
type rec_flag =
  1. | Nonrecursive
    (*

    For nonrecursive function declarations

    *)
  2. | Recursive
    (*

    For recursive function declarations

    *)
val pp_rec_flag : Ppx_deriving_runtime.Format.formatter -> rec_flag -> Ppx_deriving_runtime.unit
val show_rec_flag : rec_flag -> Ppx_deriving_runtime.string
type val_binding =
  1. | Binding of pattern * expression
    (*

    Binding(P, E) represents let P = E

    *)
and expression =
  1. | Expr_const of constant
    (*

    Constant expressions: 1, 'a', "foo", 3.14, true, 5.0<cm>

    *)
  2. | Expr_ident of string
    (*

    Identificator name expressions: x

    *)
  3. | Expr_tuple of expression list
    (*

    Tuple expressions: (E1, ..., En) Invariant: n >= 2

    *)
  4. | Expr_fun of pattern * expression
    (*

    Anonimous functions: Exp_fun(P, E) represents fun P -> E Invariant: n >= 1

    *)
  5. | Expr_let of rec_flag * val_binding list * expression option
    (*

    Expr_let(rec_flag, [(P1, E1); ...; (Pn, En)], E) represents:

    • let P1 = E1 and ... and Pn = En in E when rec_flag is Nonrecursive
    • let rec P1 = E1 and ... and Pn = En in E when rec_flag is Recursive Invariant: n >= 1
    *)
  6. | Expr_ifthenelse of expression * expression * expression option
    (*

    if E1 then E2 else E3

    *)
  7. | Expr_apply of expression * expression
    (*

    Application E1 E2

    *)
  8. | Expr_match of expression * (pattern * expression) list
    (*

    match E with P1 -> E1 | ... | Pn -> En Invariant: n >= 1

    *)
val pp_val_binding : Ppx_deriving_runtime.Format.formatter -> val_binding -> Ppx_deriving_runtime.unit
val show_val_binding : val_binding -> Ppx_deriving_runtime.string
val pp_expression : Ppx_deriving_runtime.Format.formatter -> expression -> Ppx_deriving_runtime.unit
val show_expression : expression -> Ppx_deriving_runtime.string
type structure_item =
  1. | Str_eval of expression
    (*

    Structure item which is single expression: E

    *)
  2. | Str_value of rec_flag * val_binding list
    (*

    Str_value(rec_flag, [(P1, E1); ...; (Pn, En)]) represents:

    • let P1 = E1 and ... and Pn = En when rec_flag is Nonrecursive
    • let rec P1 = E1 and ... and Pn = En when rec_flag is Recursive Invariant: n >= 1
    *)
val pp_structure_item : Ppx_deriving_runtime.Format.formatter -> structure_item -> Ppx_deriving_runtime.unit
val show_structure_item : structure_item -> Ppx_deriving_runtime.string
type program = structure_item list
val pp_program : Ppx_deriving_runtime.Format.formatter -> program -> Ppx_deriving_runtime.unit
val show_program : program -> Ppx_deriving_runtime.string
