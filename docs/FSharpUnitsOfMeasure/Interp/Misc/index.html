type error =
  1. | Division_by_zero
  2. | Type_mismatch
  3. | Unbound_identificator of string
  4. | Unsupported_operation of string
  5. | Invalid_syntax
  6. | Match_failure
  7. | Not_implemented
type builtin_fun =
  1. | Print_int of int -> unit
  2. | Print_float of float -> unit
  3. | Print_string of string -> unit
  4. | Print_char of char -> unit
  5. | Print_bool of bool -> unit
  6. | Print_endline of string -> unit
type value =
  1. | VInt of int
  2. | VFloat of float
  3. | VBool of bool
  4. | VChar of char
  5. | VString of string
  6. | VFun of Ast.rec_flag * Ast.pattern * Ast.expression * environment
  7. | VList of value list
  8. | VTuple of value * value * value list
  9. | VOption of value option
  10. | VFunction of Ast.rule * Ast.rule list
  11. | VBuiltin_fun of builtin_fun * environment
  12. | VUnit
and environment = (string, value, Base.String.comparator_witness) Base.Map.t
val is_builtin_fun : string -> bool
val pp_error : Stdlib.Format.formatter -> error -> unit
val pp_value : Stdlib.Format.formatter -> value -> unit
