type instruction =
  1. | Add of register * register * register
    (*

    Addition. rd = rs1 + rs2

    *)
  2. | Sub of register * register * register
    (*

    Subtraction. rd = rs1 - rs2

    *)
  3. | Xor of register * register * register
    (*

    Exclusive OR. rd = rs1 ^ rs2

    *)
  4. | Or of register * register * register
    (*

    OR. rd = rs1 | rs2

    *)
  5. | And of register * register * register
    (*

    AND. rd = rs1 & rs2

    *)
  6. | Sll of register * register * register
    (*

    Shift Left Logical. rd = rs1 << rs2

    *)
  7. | Srl of register * register * register
    (*

    Shift Right Logical. rd = rs1 >> rs2

    *)
  8. | Sra of register * register * register
    (*

    Shift Right Arithmetic. rd = rs1 >> rs2

    *)
  9. | Slt of register * register * register
    (*

    Set Less Than. rd = (rs1 < rs2) ? 1 : 0

    *)
  10. | Sltu of register * register * register
    (*

    Set Less Than (Unsigned)

    *)
  11. | Addi of register * register * immediate12
    (*

    Addition of Immediate. rd = rs1 + imm

    *)
  12. | Xori of register * register * immediate12
    (*

    XOR with Immediate. rd = rs1 ^ imm

    *)
  13. | Ori of register * register * immediate12
    (*

    OR with Immediate. rd = rs1 | imm

    *)
  14. | Andi of register * register * immediate12
    (*

    AND with Immediate. rd = rs1 & imm

    *)
  15. | Slli of register * register * immediate12
    (*

    Shift Left Logical with Immediate. rd = rs1 << shamt0:4

    *)
  16. | Srli of register * register * immediate12
    (*

    Shift Right Logical with Immediate. rd = rs1 >> shamt0:4 logical

    *)
  17. | Srai of register * register * immediate12
    (*

    Shift Right Arithmetic with Immediate. rd = rs1 >> shamt0:4 arithmetical

    *)
  18. | Slti of register * register * immediate12
    (*

    Set Less Than Imm. rd = (rs1 < imm) ? 1 : 0

    *)
  19. | Sltiu of register * register * immediate12
    (*

    Set Less Than Imm (Unsigned)

    *)
  20. | Lb of register * register * immediate12
    (*

    Load Byte. rd = Mrs1 + imm0:7

    *)
  21. | Lh of register * register * immediate12
    (*

    Load Half. rd = Mrs1 + imm0:15

    *)
  22. | Lw of register * register * immediate12
    (*

    Load Word. rd = Mrs1 + imm0:31

    *)
  23. | Lbu of register * register * immediate12
    (*

    Load Byte Unsigned

    *)
  24. | Lhu of register * register * immediate12
    (*

    Load Half Unsigned

    *)
  25. | Sb of register * immediate12 * register
    (*

    Store Byte. Mrs1 + imm0:7 = rs20:7

    *)
  26. | Sh of register * immediate12 * register
    (*

    Store Half. Mrs1 + imm0:15 = rs20:15

    *)
  27. | Sw of register * immediate12 * register
    (*

    Store Word. Mrs1 + imm0:31 = rs20:31

    *)
  28. | Beq of register * register * address12
    (*

    Branch ==. if (rs1 == rs2) PC += imm. PC is a program counter

    *)
  29. | Bne of register * register * address12
    (*

    Branch !=. if (rs1 != rs2) PC += imm.

    *)
  30. | Blt of register * register * address12
    (*

    Branch <. if (rs1 < rs2) PC += imm.

    *)
  31. | Bge of register * register * address12
    (*

    Branch >=. if (rs1 >= rs2) PC += imm.

    *)
  32. | Bltu of register * register * address12
    (*

    Branch < (Unsigned). if (rs1 < rs2) PC += imm.

    *)
  33. | Bgeu of register * register * address12
    (*

    Branch >= (Unsigned). if (rs1 >= rs2) PC += imm.

    *)
  34. | Jal of register * address20
    (*

    Jump and Link. rd = PC + 4; PC += imm. 4 bytes = 32 bits - instuction size

    *)
  35. | Jalr of register * register * address12
    (*

    Jump and Link Register. rd = PC + 4; PC = rs1 + imm

    *)
  36. | Lui of register * immediate20
    (*

    Load Upper Immediate. rd = imm << 12

    *)
  37. | Auipc of register * immediate20
    (*

    Add Upper Immediate to PC. rd = PC + (imm << 12)

    *)
  38. | Ecall
    (*

    EnvironmentCall - a syscall

    *)
  39. | Mul of register * register * register
    (*

    Multiply. rd = (rs1 * rs2)31:0

    *)
  40. | Mulh of register * register * register
    (*

    Multiply High. rd = (rs1 * rs2)63:32

    *)
  41. | Mulhsu of register * register * register
    (*

    Multiply High (Signed * Unsigned). rd = (rs1 * rs2)63:32

    *)
  42. | Mulhu of register * register * register
    (*

    Multiply High (Unsigned * Unsigned). rd = (rs1 * rs2)63:32

    *)
  43. | Div of register * register * register
    (*

    Division. rd = rs1 / rs2

    *)
  44. | Divu of register * register * register
    (*

    Division (Unsigned). rd = rs1 / rs2

    *)
  45. | Rem of register * register * register
    (*

    Remainder. rd = rs1 % rs2

    *)
  46. | Remu of register * register * register
    (*

    Remainder (Unsigned). rd = rs1 % rs2

    *)
  47. | Lwu of register * register * immediate12
    (*

    Load Word (Unsigned). rd = Mrs1 + imm0:31

    *)
  48. | Ld of register * register * immediate12
    (*

    Load Doubleword (Unsigned). rd = Mrs1 + imm0:63

    *)
  49. | Sd of register * register * immediate12
    (*

    Store Doubleword. Mrs1 + imm0:63 = rs20:63

    *)
  50. | Addiw of register * register * immediate12
    (*

    Addition of Immediate Word. rd = (rs1 + imm)31:0

    *)
  51. | Slliw of register * register * immediate12
    (*

    Shift Left Logical with Immediate Word. rd = (rs1 << shamt)31:0

    *)
  52. | Srliw of register * register * immediate12
    (*

    Shift Right Logical with Immediate Word. rd = (rs1 >> shamt)31:0

    *)
  53. | Sraiw of register * register * immediate12
    (*

    Shift Right Arithmetic with Immediate Word. rd = (rs1 >> shamt)31:0

    *)
  54. | Addw of register * register * register
    (*

    Add Word. rd = (rs1 + rs2)31:0

    *)
  55. | Subw of register * register * register
    (*

    Add Word. rd = (rs1 - rs2)31:0

    *)
  56. | Sllw of register * register * register
    (*

    Shifl Left Logical Word. rd = (rs1 << rs2)31:0

    *)
  57. | Srlw of register * register * register
    (*

    Shifl Right Logical Word. rd = (rs1 >> rs2)31:0

    *)
  58. | Sraw of register * register * register
    (*

    Shifl Right Arithmetical Word. rd = (rs1 >> rs2)31:0

    *)
  59. | Mulw of register * register * register
    (*

    Multiply Word. rd = (rs1 * rs2)31:0

    *)
  60. | Divw of register * register * register
    (*

    Division Word. rd = (rs1 / rs2)31:0

    *)
  61. | Divuw of register * register * register
    (*

    Division Word (Unsigned). rd = (rs1 / rs2)31:0

    *)
  62. | Remw of register * register * register
    (*

    Remainder Word. rd = (rs1 % rs2)31:0

    *)
  63. | Remwu of register * register * register
    (*

    Remainder Word (Unsigned). rd = (rs1 % rs2)31:0

    *)
  64. | Mv of register * register
    (*

    Copy from rs1 to rd. addi rd, rs1, 0

    *)
  65. | Li of register * immediate32
    (*

    Load Immediate. lui rd, immediate20; addi rd, rd, immediate12

    *)
  66. | Ret
    (*

    Return. Jalr x0, x1, 0

    *)
