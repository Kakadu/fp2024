type instruction =
  1. | Add of register * register * register
    (*

    Addition. rd = rs1 + rs2

    *)
  2. | Sub of register * register * register
    (*

    Subtraction. rd = rs1 - rs2

    *)
  3. | Xor of register * register * register
    (*

    Exclusive OR. rd = rs1 ^ rs2

    *)
  4. | Or of register * register * register
    (*

    OR. rd = rs1 | rs2

    *)
  5. | And of register * register * register
    (*

    AND. rd = rs1 & rs2

    *)
  6. | Sll of register * register * register
    (*

    Shift Left Logical. rd = rs1 << rs2

    *)
  7. | Srl of register * register * register
    (*

    Shift Right Logical. rd = rs1 >> rs2

    *)
  8. | Sra of register * register * register
    (*

    Shift Right Arithmetic. rd = rs1 >> rs2

    *)
  9. | Slt of register * register * register
    (*

    Set Less Than. rd = (rs1 < rs2) ? 1 : 0

    *)
  10. | Sltu of register * register * register
    (*

    Set Less Than (Unsigned)

    *)
  11. | Addi of register * register * address12
    (*

    Addition of Immediate. rd = rs1 + imm

    *)
  12. | Subi of register * register * address12
    (*

    Subtraction of Immediate. rd = rs1 - imm

    *)
  13. | Xori of register * register * address12
    (*

    XOR with Immediate. rd = rs1 ^ imm

    *)
  14. | Ori of register * register * address12
    (*

    OR with Immediate. rd = rs1 | imm

    *)
  15. | Andi of register * register * address12
    (*

    AND with Immediate. rd = rs1 & imm

    *)
  16. | Slli of register * register * address12
    (*

    Shift Left Logical with Immediate. rd = rs1 << shamt0:4

    *)
  17. | Srli of register * register * address12
    (*

    Shift Right Logical with Immediate. rd = rs1 >> shamt0:4 logical

    *)
  18. | Srai of register * register * address12
    (*

    Shift Right Arithmetic with Immediate. rd = rs1 >> shamt0:4 arithmetical

    *)
  19. | Slti of register * register * address12
    (*

    Set Less Than Imm. rd = (rs1 < imm) ? 1 : 0

    *)
  20. | Sltiu of register * register * address12
    (*

    Set Less Than Imm (Unsigned)

    *)
  21. | Lb of register * register * address12
    (*

    Load Byte. rd = Mrs1 + imm0:7

    *)
  22. | Lh of register * register * address12
    (*

    Load Half. rd = Mrs1 + imm0:15

    *)
  23. | Lw of register * register * address12
    (*

    Load Word. rd = Mrs1 + imm0:31

    *)
  24. | Lbu of register * register * address12
    (*

    Load Byte Unsigned

    *)
  25. | Lhu of register * register * address12
    (*

    Load Half Unsigned

    *)
  26. | Sb of register * register * address12
    (*

    Store Byte. Mrs1 + imm0:7 = rs20:7

    *)
  27. | Sh of register * register * address12
    (*

    Store Half. Mrs1 + imm0:15 = rs20:15

    *)
  28. | Sw of register * register * address12
    (*

    Store Word. Mrs1 + imm0:31 = rs20:31

    *)
  29. | Beq of register * register * address12
    (*

    Branch ==. if (rs1 == rs2) PC += imm. PC is a program counter

    *)
  30. | Beqz of register * address12
    (*

    Branch == 0. if (rs1 == 0) PC += imm. PC is a program counter

    *)
  31. | Bne of register * register * address12
    (*

    Branch !=. if (rs1 != rs2) PC += imm.

    *)
  32. | Bnez of register * address12
    (*

    Branch != 0. if (rs1 != 0) PC += imm.

    *)
  33. | Blt of register * register * address12
    (*

    Branch <. if (rs1 < rs2) PC += imm.

    *)
  34. | Bltz of register * address12
    (*

    Branch < 0. if (rs1 < 0) PC += imm.

    *)
  35. | Bgt of register * register * address12
    (*

    Branch >. if (rs1 > rs2) PC += imm.

    *)
  36. | Bgtz of register * address12
    (*

    Branch > 0. if (rs1 > 0) PC += imm.

    *)
  37. | Bge of register * register * address12
    (*

    Branch >=. if (rs1 >= rs2) PC += imm.

    *)
  38. | Bltu of register * register * address12
    (*

    Branch < (Unsigned). if (rs1 < rs2) PC += imm.

  39. | Bgeu of register * register * address12
    (*

    Branch >= (Unsigned). if (rs1 >= rs2) PC += imm.
