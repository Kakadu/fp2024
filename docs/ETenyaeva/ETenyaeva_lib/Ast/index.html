type pattern =
  1. | PatConst of const
    (*

    matches a constant value, e.g. 42, true

    *)
  2. | PatVar of id
    (*

    matches any value and binds it to a variable, e.g. x

    *)
  3. | PatAny
    (*

    matches any value without binding it - _

    *)
  4. | PatTup of pattern * pattern * pattern list
    (*

    matches tuples, e.g. (x, y), (a, b, c)

    *)
  5. | PatList of pattern list
    (*

    matches lists of patterns, e.g. y; x

    *)
  6. | PatListConstructor of pattern list
    (*

    matches lists of patterns, e.g. a::b::

    *)
  7. | PatOption of pattern option
    (*

    matches an optional pattern, e.g. Some x or None

    *)
  8. | PatWithTyp of typ * pattern
    (*

    typed pattern, e.g. a: int

    *)

type expr =
  1. | ExpVar of id
    (*

    variable, e.g. x

    *)
  2. | ExpConst of const
    (*

    constant, e.g. 10

    *)
  3. | ExpIfThenElse of expr * expr * expr option
    (*

    conditional expression, e.g. if a then b else c

    *)
  4. | ExpFun of pattern * expr
    (*

    function, e.g. fun (x, y) -> x + y

    *)
  5. | ExpFunction of case_expr * case_expr list
    (*

    function, e.g. function x | 0 -> "zero" | _ -> "nonzero"

    *)
  6. | ExpBinOper of binary_oper * expr * expr
    (*

    binary operation, e.g. 1 + 5

    *)
  7. | ExpUnOper of unary_oper * expr
    (*

    unary operation, e.g. -7

    *)
  8. | ExpList of expr list
    (*

    list expression, e.g. 1, "string", 2, (1 + 7)

    *)
  9. | ExpListConstructor of expr list
    (*

    list expression, e.g. 1::2::

    *)
  10. | ExpLet of rec_flag * let_binding * let_binding list * expr
    (*

    let, e.g. let x = 5

    *)
  11. | ExpApp of expr * expr
    (*

    application, e.g. (fun (x, y) -> x + y) (1, 2)

    *)
  12. | ExpTup of expr * expr * expr list
    (*

    tuple expression, e.g. (e1, e2), (x, y, z)

    *)
  13. | ExpMatch of expr * case_expr * case_expr list
    (*

    pattern matching, e.g. match x with | 0 -> "zero" | _ -> "nonzero"

    *)
  14. | ExpOption of expr option
    (*

    optonal expression, e.g. Some x

    *)
  15. | ExpWithTyp of typ * expr
    (*

    typed expression, e.g. a: int

    *)
and let_binding = {
  1. pat : pattern;
    (*

    the pattern being bound, e.g. x, (a, b)

    *)
  2. expr : expr;
    (*

    the expression being assigned, e.g. 42, fun x -> x + 1

    *)
}
and case_expr = {
  1. case_pat : pattern;
    (*

    the pattern to match, e.g. x, _

    *)
  2. case_expr : expr;
    (*

    the expression to evaluate if the pattern matches

    *)
}
