type comprehension =
  1. | Condition of expr
    (*

    e.g. x < 10

    *)
  2. | Generator of pattern * expr
    (*

    e.g. x <- [1 ..]

    *)
and ordinarylistbld =
  1. | ComprehensionList of expr * comprehension * comprehension list
    (*

    e.g [x * y | x <- [1, 10, 100], y <- [2, 3]], y mod 2 == 0

    *)
  2. | IncomprehensionlList of expr
    (*

    e.g. [1,2]

    *)
and listbld =
  1. | LazyList of expr * expr option * expr option
    (*

    e.g. [1.. ] or [1..10] or [1, 2 .. 10] or [1, 3..]

    *)
  2. | OrdList of ordinarylistbld
and binding =
  1. | VarsBind of pattern * bindingbody * binding list
    (*

    e.g x = let y = 12 in y * z where z = 5

    *)
  2. | FunBind of ident * functype option * pattern * pattern list * bindingbody * binding list
    (*

    e.g f x y = x + y + z where z = 2

    *)
and bindingbody =
  1. | Guards of expr * expr * (expr * expr) list
    (*

    (condition, branch) pairs e.g f x | x > 0 = x | otherwise = -1

    *)
  2. | OrdBody of expr
    (*

    e.g f x = if x > 0 then x else -1

    *)

examples below are for function binding with due body

and binary_tree_bld =
  1. | Nul
    (*

    node that not exists (notation: $)

    *)
  2. | Node of expr * binary_tree_bld * binary_tree_bld
    (*

    node is data and two 'sons' e.g (x^y; $; (2; $; $))

    *)
and expression =
  1. | Const of const
  2. | Identificator of ident
    (*

    e.g x

    *)
  3. | TupleBld of expr * expr * expr list
    (*

    e.g (1+3, f x)

    *)
  4. | OptionBld of expr maybe
    (*

    e.g Just (f x)

    *)
  5. | ListBld of listbld
    (*

    e.g [(2 ^ 2 - 3) ..]

    *)
  6. | Binop of expr * binop * expr
    (*

    e.g 1 > 0

    *)
  7. | Unop of unop * expr
    (*

    e.g (-1)

    *)
  8. | IfThenEsle of expr * expr * expr
    (*

    e.g if x >= 0 then x else (-x)

    *)
  9. | FunctionApply of expr * expr * expr list
    (*

    e.g. sum 1 2 or \x -> x + 1) 1

    *)
  10. | Lambda of pattern * pattern list * expr
    (*

    e.g. \x y -> x + y

    *)
  11. | BinTreeBld of binary_tree_bld
  12. | Case of pattern * pattern * expr * (pattern * expr) list
    (*

    e.g case l of (x:xs) -> x [] -> 0

    *)
  13. | InnerBindings of binding * binding list * expr
    (*

    e.g. let x = 1 y = 2 in x + y

    *)
and expr = expression * tp option

e.g. (x + 1) :: Int
