type comprehension =
  1. | Condition of expr
    (*

    e.g. x < 2

    *)
  2. | Generator of pattern * expr
    (*

    e.g. x <- [1 ..]

    *)
and ordinarylistbld =
  1. | IncomprehensionlList of expr_list
    (*

    e.g. [1,2]

    *)
and listbld =
  1. | LazyList of expr * expr option * expr option
    (*

    e.g. [1.. ] or [1..2] or [1, 2 .. 2] or [1, 3..]

    *)
  2. | OrdList of ordinarylistbld
and def =
  1. | VarsDef of pattern * bindingbody * binding_list
    (*

    e.g x = let y = 12 in y * z where z = 5

    *)
  2. | FunDef of ident * pattern * pattern_list * bindingbody * binding_list
    (*

    e.g f x y = x + y + z where z = 2

    *)
and binding =
  1. | Def of def
  2. | Decl of ident * tp
    (*

    e.g f :: Int -> Int

    *)
and binding_list = binding list
and pattern_bindinbody_list = (pattern * bindingbody) list
and bindingbody =
  1. | Guards of expr * expr * expr_expr_list
    (*

    (condition, branch) pairs e.g f x | x > 0 = x | otherwise = -1

    *)
  2. | OrdBody of expr
    (*

    e.g f x = if x > 0 then x else -1

    *)

examples below are for function binding with due body

and binary_tree_bld =
  1. | Nul
    (*

    node that not exists (notation: $)

    *)
  2. | Node of expr * expr * expr
    (*

    node is data and two 'sons' e.g (x^y; $; (2; $; $))

    *)
and expression =
  1. | Const of const
  2. | Identificator of ident
    (*

    e.g x

    *)
  3. | TupleBld of expr * expr * expr_list
    (*

    e.g (1+3, f x)

    *)
  4. | EJust
    (*

    Maybe constructor Just

    *)
  5. | ENothing
  6. | ListBld of listbld
    (*

    e.g [(2 ^ 2 - 3) ..]

    *)
  7. | Binop of expr * binop * expr
    (*

    e.g 1 > 0

    *)
  8. | Neg of expr
    (*

    e.g (-1)

    *)
  9. | IfThenEsle of expr * expr * expr
    (*

    e.g if x >= 0 then x else (-x)

    *)
  10. | FunctionApply of expr * expr * expr_list
    (*

    e.g. sum 1 2 or \x -> x + 1) 1

    *)
  11. | Lambda of pattern * pattern_list * expr
    (*

    e.g. \x y -> x + y

    *)
  12. | BinTreeBld of binary_tree_bld
  13. | Case of expr * pattern * bindingbody * pattern_bindinbody_list
    (*

    e.g case l of (x:xs) -> x; [] -> 0

    *)
  14. | InnerBindings of binding * binding_list * expr
    (*

    e.g. let x = 1; y = 2 in x + y

    *)
and expr = expression * tp_list

e.g. ((x + 1) :: Int ) :: Bool

and expr_list = expr list
and expr_expr_list = (expr * expr) list
