open Miniml.Ast

let factorial_ast : program =
  (* View of program:
     let rec factorial n = if (n > 1) then n * factorial(n-1) else 1;;
     factorial 5;;
  *)
  [ DefineItem
      ( Recursive
      , [ ( PVar "factorial"
          , Lambda
              ( [ PVar "n" ]
              , If
                  ( Binary (Variable "n", Gt, Const (IntLiteral 1))
                  , Binary (Variable "n", Multiply, Const (IntLiteral 1))
                  , Some (Const (IntLiteral 1)) ) ) )
        ] )
  ; EvalItem (Apply (Variable "factorial", [ Const (IntLiteral 5) ]))
  ]
;;

let%expect_test _ =
  Format.printf "%s" (show_program factorial_ast);
  [%expect
    {|
    [(DefineItem
        (Recursive,
         [((PVar "factorial"),
           (Lambda ([(PVar "n")],
              (If ((Binary ((Variable "n"), Gt, (Const (IntLiteral 1)))),
                 (Binary ((Variable "n"), Multiply, (Const (IntLiteral 1)))),
                 (Some (Const (IntLiteral 1)))))
              )))
           ]));
      (EvalItem (Apply ((Variable "factorial"), [(Const (IntLiteral 5))])))] |}]
;;
